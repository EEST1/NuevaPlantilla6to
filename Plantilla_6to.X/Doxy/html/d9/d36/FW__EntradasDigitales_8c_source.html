95 {

97  {
98  uint8_t cambios = 0, j = 0;
99  // cambios me avisa si hay algun cambio, esto indica que hay alguna tecla presionada
100  // en J se guarda el estado de cada tecla, 1 bit por cada tecla
101  static uint8_t contador_estados[ED_ENTRADAS];
102  // aca se guarda el estado de la cuenta para validar la tecla
103 
104  j = !BOT1;
105  j |= (uint8_t) (!BOT2 << (0x01));
106  j |= (uint8_t) (!BOT3 << (0x02));
107  j |= (uint8_t) (!BOT4 << (0x03));
108 
109  cambios = ED_BufferEntradas ^ j ; // en cambios queda bit por bit los cambios registrados
110  // correspondentes a cada tecla precionada;
111 
112  if(cambios)
113  {
114  for(j=0; j < ED_ENTRADAS; j++)
115  {
116  if(cambios & (0x01) << j)
117  {
118  contador_estados[j]++;
119  if(contador_estados[j]== ED_ACEPTAR_ESTADO)
120  {
121  ED_BufferEntradas ^= (0x01 << j);
122  contador_estados[j] = 0;
123  }
124  }
125  else
126  contador_estados[j] = 0;
127  }
128  }
129  else
130  {
131 
132  for(j=0; j < ED_ENTRADAS; j++)
133  {
134  contador_estados[j] = 0;
135  }
136  }

138  }
